## 安装

<Steps>
### 安装依赖

安装 Rabbykit 及其依赖

```sh npm2yarn
npm i @rabby-wallet/rabbykit @wagmi/core viem
```

### 使用 Rabbykit 和 wagmi

配置好`wagmi`，然后设置`rabbykit`

```ts filename="app.ts"
import { configureChains, createConfig, mainnet } from "@wagmi/core";
import { mainnet, arbitrum, bsc, optimism, polygon } from "@wagmi/core/chains";
import { createModal } from "@rabby-wallet/rabbykit";

const { chains, publicClient, webSocketPublicClient } = configureChains(
  [mainnet, arbitrum, bsc, optimism, polygon],
  [
    alchemyProvider({ apiKey: "yourAlchemyApiKey" }),
    infuraProvider({ apiKey: "yourInfuraApiKey" }),
    publicProvider(),
  ]
);

const config = createConfig({
  autoConnect: true,
  publicClient,
  webSocketPublicClient,
});

export const rabbyKit = createModal({
  chains,
  wagmi: config,
  projectId: "yourProjectId",
  appName: "RabbyKit",
});
//打开钱包连接弹窗
rabbykit.open();

console.log("current rabbykit modal open status:", rabbyKit.getOpenState());

//关闭钱包连接弹窗
rabbykit.close();
```

</Steps>
